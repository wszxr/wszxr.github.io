二叉树链式存储

初始化

//节点类，包含数据，左右孩子指针
class TreeNode{
    int data;
    TreeNode left;
    TreeNode right;
}

//二叉树类中手动初始化一棵树
public void init(){
    root = new TreeNode(1);
    root.left = new TreeNode(2);
    root.right = new TreeNode(3);
    root.left.left = new TreeNode(4);
    root.left.right = new TreeNode(5);
    root.right.left = new TreeNode(6);
    root.right.right = new TreeNode(7);
}

遍历

//前序遍历，中左右顺序
public void frontShow(){
    System.out.println(data);
    if(left!=null){
        left.frontShow();
    }
    if(right!=null){
        right.frontShow();
    }
}

//中序遍历，左中右顺序
public void midShow(){
    if(left!=null){
        left.midShow();
    }
    System.out.println(data);
    if(right!=null){
        right.midShow();
    }
}

//后序遍历，左右中顺序
public void afterShow(){
    if(left!=null){
        left.afterShow();
    }
    if(right!=null){
        right.afterShow();
    }
    System.out.println(data);
}

查找

//三种遍历方法对应三种查找,传入目标值，返回节点对象

//前序查找
public TreeNode frontSearch(int target){
    //当temp不为空说明已经查到，没有继续的意义
    TreeNode temp = null;
    if(target==this.data){
        return this;
    }else {
        if(left!=null){
            temp =  left.frontSearch(target);
        }
        if(temp !=null){
            return  temp;
        }

        if(right!=null){
            temp= right.frontSearch(target);
        }

    }
    return temp;

}

//中序查找
public TreeNode midSearch(int target){
    TreeNode temp = null;
    if(left!=null){
        temp = left.midSearch(target);
    }
    if(temp!=null){
        return temp;
    }

    if(this.data==target){
        return this;
    }
    if(right!=null){
        temp = right.midSearch(target);
    }
    return temp;

}

//后序查找
public TreeNode afterSearch(int target){
    TreeNode temp = null;
    if (left!=null){
        temp = left.afterSearch(target);
    }
    if(temp!=null){
        return temp;
    }
    if (right!=null){
        temp = right.afterSearch(target);
    }
    if(temp!=null){
        return temp;
    }
    if(this.data==target){
        return this;
    }
    return temp;
}

删除

//这是节点类中的方法

//删除节点，
public void delete(int target){
    //子节点满足条件就断掉
    if(left!=null&&left.data==target){
        left = null;
        return;
    }
    if(right!=null&&right.data==target){
        right = null;
        return;
    }
    //遍历除了叶子节点之外的所有节点
    if(left!=null){
        left.delete(target);
    }
    if(right!=null){
        right.delete(target);
    }

}

//二叉树类的方法，调用节点类中的方法并处理空树和只有一个根节点的情况

//删除节点
public void delete(int target){
    //根节点没有双亲，无法删除，单独判断
    if(root==null){
        throw new RuntimeException("树空");
    }else if(root.data==target){
        root = null;
    }else {
        //调用节点类方法
        root.delete(target);
    }
}

二叉树顺序存储

/*
1，index是下标

2，2*index+1表示左孩子，2*index+2表示右孩子

3，(n-1)2是双亲节点

4，顺序二叉树通常只考虑完全二叉树
*/

//只需要一个数组即可
class ArrayBinaryTree {
    //存储树的数组
    int []array;
    //相当于传一棵树进来
    public ArrayBinaryTree(int []array){
        this.array = array;
    }
}

遍历

//前序遍历

public  void frontShow(int index){
    //空数组或是数组为0无法遍历
    if(array==null||array.length==0){
        throw new RuntimeException("无法遍历");
    }
    //打印出当前值
    System.out.println(array[index]);
    //存在左节点就向左递归
    if(2*index+1<array.length){
        frontShow(2*index+1);
    }
    //存在右节点就向右递归
    if(2*index+2<array.length){
        frontShow(2*index+2);
    }
}

//中序遍历
public void midShow(int index){
    if(array==null||array.length==0){
        throw new RuntimeException("无法遍历");
    }
    if(index*2+1<array.length){
        midShow(index*2+1);
    }
    System.out.println(array[index]);
    if(index*2+2<array.length){
        midShow(index*2+2);
    }
}

//后序遍历
public void afterShow(int index){
    if(array==null||array.length==0){
        throw new RuntimeException("无法遍历");
    }

    if(2*index+1<array.length){
        afterShow(2*index+1);
    }
    if(2*index+2<array.length){
        afterShow(2*index+2);
    }
    System.out.println(array[index]);
}

线索二叉树

有些节点指针域为空，浪费了空间

//线索化二叉树，中序
public void threadTree(TreeNode node){
    //递归往最左边找，找到叶子节点的下一个结束
    if(node==null){
        return;
    }
    //递归左子树
    threadTree(node.left);
    //左子树为空指向前驱节点
    if(node.left==null){
        node.left = pre;
        //leftType==1证明他的左子树指向前驱节点
        node.leftType = 1;
    }
    if(pre!=null&&pre.right ==null){
        pre.right = node;
        //rightType == 1证明右子树指向后继节点
        pre.rightType = 1;
    }

    //left向递归遍历顺序后移一个
    pre = node;
    //递归右子树
    threadTree(node.right);
}

哈夫曼树

public static Node hfm(int [] row){
        List<Node> list = new ArrayList<>();
        for (int i : row) {
            list.add(new Node(i));
        }
        //当列表中只剩下目标树的根节点时停止循环
        while(list.size()>1){
            //排序，需要list中的元素实现Comparable接口
            Collections.sort(list);
            //把权值最小的两个取出当第一棵子树的左右孩子节点
            Node leftNode = list.get(0);
            Node rightNode = list.get(1);
            //双亲节点的权值是两孩子节点的权值和
            Node parent = new Node(leftNode.value+rightNode.value);
            //把孩子节点和双亲节点连起来
            parent.left = leftNode;
            parent.right = rightNode;
            //移除列表中的孩子节点
            list.remove(leftNode);
            list.remove(rightNode);
            //把双亲节点加入
            list.add(parent);
        }
        //返回哈夫曼树的根节点
       return list.get(0);
    }
