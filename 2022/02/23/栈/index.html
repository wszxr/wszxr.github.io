栈篇

发表于 2022-02-23 更新于 2022-02-26 分类于 数据结构

栈是一种后进先出结构(FILO)

用数组实现栈

任何实现入栈出栈时间复杂度都是O(1)
class Stack{
private int maxSize;
    private int []datas ;
    private int top = -1;//栈顶初始值为-1
    public Stack(int maxSize){
        this.maxSize = maxSize;
        datas = new int[maxSize];
    }

//入栈
public void push(int data){
    if(isFull()){
        throw new RuntimeException("栈满了无法加入");
    }
    top++;//栈顶先移动
    datas[top] = data;
}

//出栈
public int pop(){
    if(isEmpty()) {
        throw new RuntimeException("栈空无法出栈");
    }
    int temp = datas[top];
    top--;
    return temp;
}

用链表实现

//链表实现不用担心栈满的问题
class LinkedStack {
Node head = new Node();
Node top;
}

//入栈，头节点为栈底，先连接头节点，在把top指向入栈的
public void push(Node node){
    node.next = top;
    top = node;
}

//出栈
public int  pop(){
    if(isEmpty()){
        throw new RuntimeException("栈空无法出栈");
    }
    int a = top.data;
    top = top.next;
    return a;
}

共享栈

//特定环境使用，一个数组中放两个栈，左边初始top1 = -1，右边初始top2 = maxSize.
//两个栈应该类型相同且一个栈增加的同时另一个栈相应减小，否则很快也会满

class ShareStack {
    private int maxSize;
    private int[]datas;
    private int top1;
    private int top2;
}

//判断栈满,top1,top2相邻
public boolean isFull(){
    return top1+1==top2;
}

//判断栈空，分别判断
public boolean isEmpty(int number){
    if(number==1){
        return top1==-1;
    }else return top2==maxSize;
}


//进栈，需要指定哪一个栈
public void push(int data,int number){
    //满了抛出异常
    if(isFull()){
        throw new RuntimeException("栈满无法加入");
    }
    //当number==1时表示加到左边的栈，否则加到右边的栈
    if(number==1){
        top1++;
        datas[top1] = data;
    }else {
        top2--;
        datas[top2] = data;
    }
}

//出栈，需要指定哪一个栈
public int pop(int number){
    //判断是否为空
    if(isEmpty(number)){
        throw new RuntimeException("栈空无法出栈");
    }

    if(number==1){
        int l = datas[top1];
        top1--;
        return l;
    }else {
        int r = datas[top2];
        top2--;
        return r;
    }
}
