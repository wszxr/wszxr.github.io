二叉树链式存储

初始化

//节点类，包含数据，左右孩子指针
class TreeNode{
    int data;
    TreeNode left;
    TreeNode right;
}

//二叉树类中手动初始化一棵树
public void init(){
    root = new TreeNode(1);
    root.left = new TreeNode(2);
    root.right = new TreeNode(3);
    root.left.left = new TreeNode(4);
    root.left.right = new TreeNode(5);
    root.right.left = new TreeNode(6);
    root.right.right = new TreeNode(7);
}

遍历

//遍历，需要传入开始节点

//前序遍历，中左右顺序
public void frontShow(){
    System.out.println(data);
    if(left!=null){
        left.frontShow();
    }
    if(right!=null){
        right.frontShow();
    }
}

//中序遍历，左中右顺序
public void midShow(){
    if(left!=null){
        left.midShow();
    }
    System.out.println(data);
    if(right!=null){
        right.midShow();
    }
}

//后序遍历，左右中顺序
public void afterShow(){
    if(left!=null){
        left.afterShow();
    }
    if(right!=null){
        right.afterShow();
    }
    System.out.println(data);
}

查找

//三种遍历方法对应三种查找,传入目标值，返回节点对象

//前序查找
public TreeNode frontSearch(int target){
    //当temp不为空说明已经查到，没有继续的意义
    TreeNode temp = null;
    if(target==this.data){
        return this;
    }else {
        if(left!=null){
            temp =  left.frontSearch(target);
        }
        if(temp !=null){
            return  temp;
        }

        if(right!=null){
            temp= right.frontSearch(target);
        }

    }
    return temp;

}

//中序查找
public TreeNode midSearch(int target){
    TreeNode temp = null;
    if(left!=null){
        temp = left.midSearch(target);
    }
    if(temp!=null){
        return temp;
    }

    if(this.data==target){
        return this;
    }
    if(right!=null){
        temp = right.midSearch(target);
    }
    return temp;

}

//后序查找
public TreeNode afterSearch(int target){
    TreeNode temp = null;
    if (left!=null){
        temp = left.afterSearch(target);
    }
    if(temp!=null){
        return temp;
    }
    if (right!=null){
        temp = right.afterSearch(target);
    }
    if(temp!=null){
        return temp;
    }
    if(this.data==target){
        return this;
    }
    return temp;
}

删除

//这是节点类中的方法

//删除节点，
public void delete(int target){
    //子节点满足条件就断掉
    if(left!=null&&left.data==target){
        left = null;
        return;
    }
    if(right!=null&&right.data==target){
        right = null;
        return;
    }
    //遍历除了叶子节点之外的所有节点
    if(left!=null){
        left.delete(target);
    }
    if(right!=null){
        right.delete(target);
    }

}

//二叉树类的方法，调用节点类中的方法并处理空树和只有一个根节点的情况

//删除节点
public void delete(int target){
    //根节点没有双亲，无法删除，单独判断
    if(root==null){
        throw new RuntimeException("树空");
    }else if(root.data==target){
        root = null;
    }else {
        //调用节点类方法
        root.delete(target);
    }
}

普通

//每次出队列都需要移动所有的元素，解决方法，循环队列。

循环队列

//解决了移动元素的问题，用头尾指针的移动替代了。

//成员变量
class Queue {
    private int maxSize;
    private int []datas;
    /*
    front 指向队列头
    rear 指向队列尾的后一个元素
     */
    int front;
    int rear;
}

//判断队列是否空，头和尾重合即为空
public boolean isEmpty(){
    return front==rear;
}
//判断队列是不是满，尾指针一直指向空
public boolean isFull(){
    return (rear+1)%maxSize==front;
}

//入队列操作
public void enter(int data){
    if(isFull()){
        throw new RuntimeException("队列满了无法加入");
    }
    datas[rear] = data;//值付给队尾指针
    rear = (rear+1)%maxSize;//队尾加一，超过循环后取余maxSize
}

//出队列操作
public int  delete(){
    if(isEmpty()){
        throw new RuntimeException("队列空无法出队");
    }
    int num = datas[front];
    front = (front+1)%maxSize;
    return num;
}

/*如果头在尾巴的前面，长度就是尾减去头，
  如果头在尾后面，长度为max-front这一段加上rear-0这一段
*/
//长度
public int length(){
    return (maxSize+rear-front)%maxSize;
}

用链表实现

//不用考虑溢出问题

//成员变量，需要一个头节点
public class LinkQueue {
    Node front;
    Node rear;
    Node head;
}

//添加节点，先连接上下一个节点，在移动rear指针。
public void enter(Node node){
    rear.next = node;
    rear = rear.next;
}

//删除节点
public int delete(){
    if(isEmpty()){//头和尾重合即为空
        throw new RuntimeException("队列空无法出队列");
    }
    int a = front.next.data;

    //如果要删除的节点是最后一个，为了不让rear指针指空，把rear放回头节点
    if(front.next == rear){
        rear = head;
    }
    //跨过头结点的下一个节点即可
    front.next = front.next.next;

    return a;
}

栈是一种后进先出结构(FILO)

用数组实现栈

任何实现入栈出栈时间复杂度都是O(1)
class Stack{
private int maxSize;
    private int []datas ;
    private int top = -1;//栈顶初始值为-1
    public Stack(int maxSize){
        this.maxSize = maxSize;
        datas = new int[maxSize];
    }

//入栈
public void push(int data){
    if(isFull()){
        throw new RuntimeException("栈满了无法加入");
    }
    top++;//栈顶先移动
    datas[top] = data;
}

//出栈
public int pop(){
    if(isEmpty()) {
        throw new RuntimeException("栈空无法出栈");
    }
    int temp = datas[top];
    top--;
    return temp;
}

用链表实现

//链表实现不用担心栈满的问题
class LinkedStack {
Node head = new Node();
Node top;
}

//入栈，头节点为栈底，先连接头节点，在把top指向入栈的
public void push(Node node){
    node.next = top;
    top = node;
}

//出栈
public int  pop(){
    if(isEmpty()){
        throw new RuntimeException("栈空无法出栈");
    }
    int a = top.data;
    top = top.next;
    return a;
}

共享栈

//特定环境使用，一个数组中放两个栈，左边初始top1 = -1，右边初始top2 = maxSize.
//两个栈应该类型相同且一个栈增加的同时另一个栈相应减小，否则很快也会满

class ShareStack {
    private int maxSize;
    private int[]datas;
    private int top1;
    private int top2;
}

//判断栈满,top1,top2相邻
public boolean isFull(){
    return top1+1==top2;
}

//判断栈空，分别判断
public boolean isEmpty(int number){
    if(number==1){
        return top1==-1;
    }else return top2==maxSize;
}


//进栈，需要指定哪一个栈
public void push(int data,int number){
    //满了抛出异常
    if(isFull()){
        throw new RuntimeException("栈满无法加入");
    }
    //当number==1时表示加到左边的栈，否则加到右边的栈
    if(number==1){
        top1++;
        datas[top1] = data;
    }else {
        top2--;
        datas[top2] = data;
    }
}

//出栈，需要指定哪一个栈
public int pop(int number){
    //判断是否为空
    if(isEmpty(number)){
        throw new RuntimeException("栈空无法出栈");
    }

    if(number==1){
        int l = datas[top1];
        top1--;
        return l;
    }else {
        int r = datas[top2];
        top2--;
        return r;
    }
}

链表和顺序表的优劣

1,因为有数组下标的存在，顺序表在查找方面领先链表，时间复杂度为O(1),而链表查找需要遍历，因此时间复杂度为o(N)。

2，增加或删除元素方面，链表优势，顺序表每次插入或者删除都需要移动大量元素O(N),链表需要查找到增加或删除的位置(通过指针)，也是O(N),但是插入的越频繁链表的优势体现的越大(指针移动的距离变少)。

双向链表

1,除了data和next外，还有个指向前驱节点的prior指针。

2,空间占用更多。用空间换时间效率。

插入操作：//找到插入位置的前一个位置temp
        node.next = temp.next;//设置待插入节点的next
        node.prior = temp;//设置待插入节点的prior
        temp.next.prior = node;//将待插入节点的下一个节点的prior指向待插入节点
        temp.next = node;//将temp指向待插入节点

删除操作：//找到待删除的节点temp
        //直接跳过删除节点
        temp.prior.next = temp.next
        temp.next.prior = temp.prior;
